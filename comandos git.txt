PUSH
git push -u origin --all

FETCH
git fetch origin

REBASE
git rebase origin/master

BRANCH
[local]
- crear y seleccionar
git checkout –b <nombre rama local> origin/master

- eliminar
git branch -d <nombre rama local>

- modificar

- seleccionar
git checkout <nombre rama local>

[remota]
- crear

- eliminar
git push origin :<nombre rama remota>

- modificar
